# Province adopts Norman culture
namespace = CK2RPG
province_event = {
	id = CK2RPG.0
	desc = EVTDESC_NEOGAULFLIP
	picture = GFX_evt_throne_room
	
	trigger = {
		OR = {
			culture = frankish
			culture = breton
			culture = occitan
		}
		owner = {
			top_liege = { culture_group = Celtic }
		}
	}
	
	mean_time_to_happen = {
		months = 1200
	}

	option = {
		name = EVTOPTA_TOG_2003
		trigger = {
			NOT = {
				owner = { culture = norman }
			}
		}
		culture = norman
	}
	option = {
		name = EVTOPTB_TOG_2003
		trigger = {
			owner = { culture = norman }
		}
		culture = norman
	}
}
