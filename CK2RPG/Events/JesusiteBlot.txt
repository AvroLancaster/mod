# You summon your vassals to a Blot
character_event = {
	id = CK2RPGBLOT.600
	desc = EVTDESC_CK2RPGBLOT_600
	picture = GFX_evt_viking_throneroom_oldgods
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	hide_from = yes
	
	option = {
		name = EVTOPTA_CK2RPGBLOT_600
		any_vassal = {
			limit = {
				religion = jesusite
				age = 16
				prisoner = no
				war = no
				NOT = { trait = incapable }
			}
			character_event = { id = CK2RPGBLOT.601 tooltip = EVTTOOLTIP_TOG_601 }
			hidden_tooltip = { character_event = { id = CK2RPGBLOT.605 days = 10 } }
		}
		any_vassal = {
			limit = {
				NOT = {
					religion = jesusite
				}
				age = 16
				prisoner = no
				war = no
				NOT = { trait = incapable }
			}
			character_event = { id = CK2RPGBLOT.602 tooltip = EVTTOOLTIP_TOG_602 }
		}
		hidden_tooltip = { character_event = { id = CK2RPGBLOT.604 days = 10 } }
	}
}

# Non-Jesusites pay to skip Blot
character_event = {
	id = CK2RPGBLOT.602
	desc = EVTDESC_CK2RPGBLOT_602
	picture = GFX_evt_viking_throneroom_oldgods
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	hide_from = yes
	
	option = { # Pay
		name = EVTOPTA_TOG_602
		trigger = {
			wealth = 20
		}
		FROM = {
			character_event = { id = CK2RPGBLOT.603 tooltip = EVTTOOLTIP_TOG_603 }
		}
		wealth = -20
	}
	option = { # Don't pay
		name = EVTOPTB_TOG_602
		FROM = {
			character_event = { id = CK2RPGBLOT.636 tooltip = EVTTOOLTIP_TOG_636 }
		}
		tooltip = {
			FROM = {
				opinion = {
					modifier = opinion_refused_blot
					who = ROOT
				}
				any_vassal = {
					limit = { has_character_flag = attending_blot }
					opinion = {
						modifier = opinion_refused_blot
						who = ROOT
					}
				}
			}
		}
	}
}

	id = CK2RPGBLOT.660
	hide_window = yes
	
	trigger = {
		has_dlc = "The Old Gods"
		has_character_flag = blot_feast_vassal
		NOT = { has_character_flag = feast_friendship }
		liege = { 
			has_character_flag = blot_feast
			any_vassal = {
				has_character_flag = blot_feast_vassal
				NOT = { has_character_flag = feast_friendship }
				NOT = { character = ROOT }
				opinion = { who = ROOT value =  20 }
			}		
		}
	}
	
	mean_time_to_happen = {
		days = 60
		
		modifier = {
			factor = 1.1
			liege = { num_of_vassals = 10 }
		}
		modifier = {
			factor = 1.2
			liege = { num_of_vassals = 20 }
		}
		modifier = {
			factor = 1.3
			liege = { num_of_vassals = 30 }
		}
		modifier = {
			factor = 1.4
			liege = { num_of_vassals = 40 }
		}
		modifier = {
			factor = 1.5
			liege = { num_of_vassals = 50 }
		}
		modifier = {
			factor = 3
			liege = { num_of_vassals = 100 }
		}
	}
	
	immediate = {
		set_character_flag = feast_friendship
		liege = { 
			random_vassal = {
				limit = {
					has_character_flag = blot_feast_vassal
					NOT = { character = ROOT }
					opinion = { who = ROOT value =  20 }
				}
				character_event = { id = CK2RPGBLOT.661 }
				set_character_flag = feast_friendship
				opinion = {
					modifier = opinion_feast_friend
					who = ROOT
				}
				reverse_opinion = {
					modifier = opinion_feast_friend
					who = ROOT
				}
			}		
		}
	}
	
	option = {
		name = OK
	}
}

					}
				}

					limit = {

						trait = homosexual
					}
					opinion = {
						modifier = opinion_homosexual_sympathy

				}
			}
		}	
	}
